# ISO 532-1 Rust + AVX 實作計畫

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 將 mosqito 1.2.1 的 ISO 532-1:2017 響度計算（穩態 zwst + 時變 zwtv）改寫為純 Rust crate `iso532`，熱點提供 AVX2+FMA 加速（runtime 偵測、scalar fallback），以 mosqito 逐階段 golden 資料與 ISO Annex B 官方測試訊號雙層驗證。

**Architecture:** scalar 版照 ISO C 參考程式的逐 frame 邏輯實作（非 mosqito 的 numpy mask 向量化）；SIMD 平行軸為「跨頻帶」（28 頻帶 = 7 個 f64×4）。因輸入固定 48 kHz，zwst 的 Butterworth/Cheby1 濾波器係數由 `tools/gen_tables.py` 用 scipy 生成後烘焙為 Rust 常數（消除重現 scipy 濾波器設計的數值風險）；zwtv 濾波器係數直接來自 ISO Table A.1/A.2。

**Tech Stack:** Rust 2021（deps: `thiserror`；dev-deps: `criterion`, `hound`）、`std::arch` AVX2/FMA intrinsics、Python venv（mosqito 1.2.1 + scipy）產生 golden 與係數。

**Spec:** `docs/superpowers/specs/2026-07-03-iso532-1-rust-avx-design.md`

**參考來源（唯讀）：** `D:\ISO532\mosqito-1.2.1\`（已解壓，不入 git）

**慣例：**
- 工作目錄：repo 根 `D:\ISO532`，crate 在 `iso532/`。shell 指令用 Git Bash 語法。
- 所有 golden 比對容差：結構相同處 `rtol=1e-9`，濾波/冪運算鏈 `rtol=1e-7, atol=1e-12`（任務內另有指定者從其指定）。
- `data/` 全部 gitignore（可由工具重生），工具腳本入 git。
- 每個 golden 測試失敗時：先檢查 tables 轉錄錯字，再檢查比較運算子（`<` vs `<=`）與 mosqito 的 `round(x,8)` 容差語意。

- [ ] **Step 3: 寫 gen_golden.py（完整內容）**

```python
"""Generate per-stage golden data from mosqito 1.2.1 for the Rust port.

Output layout: data/golden/<name>/<stage>.bin  (little-endian f64, C order)
plus meta.json with array shapes.
Run: .venv/Scripts/python tools/gen_golden.py
"""
import json
from pathlib import Path

import numpy as np
from scipy.signal import cheby1, sosfilt, sosfiltfilt, decimate

from mosqito.sound_level_meter import noct_spectrum
from mosqito.utils import amp2db
from mosqito.utils import load
from mosqito.sq_metrics.loudness.loudness_zwst._main_loudness import _main_loudness
from mosqito.sq_metrics.loudness.loudness_zwst._calc_slopes import _calc_slopes
from mosqito.sq_metrics.loudness.loudness_zwtv._third_octave_levels import _third_octave_levels
from mosqito.sq_metrics.loudness.loudness_zwtv._nonlinear_decay import _nl_loudness
from mosqito.sq_metrics.loudness.loudness_zwtv._temporal_weighting import _temporal_weighting

FS = 48000
ROOT = Path(__file__).resolve().parent.parent
OUT = ROOT / "data" / "golden"


def dump(d: Path, name: str, arr):
    arr = np.ascontiguousarray(np.asarray(arr, dtype="<f8"))
    arr.tofile(d / f"{name}.bin")
    return list(arr.shape)


def sine(freq, db, dur):
    t = np.arange(0, dur, 1 / FS)
    p_rms = 2e-5 * 10 ** (db / 20)
    return np.sqrt(2) * p_rms * np.sin(2 * np.pi * freq * t)


def make_signals():
    rng = np.random.default_rng(532)
    sigs = {
        "sine_1k_60": sine(1000, 60, 1.0),
        "sine_250_80": sine(250, 80, 1.0),
        "sine_4k_60": sine(4000, 60, 1.0),
        "white_60": None,
        "pulse_1k_70": None,
        "step_60_80": None,
    }
    w = rng.standard_normal(FS)
    w *= (2e-5 * 10 ** (60 / 20)) / np.sqrt(np.mean(w**2))
    sigs["white_60"] = w
    p = np.zeros(FS)
    burst = sine(1000, 70, 0.01)
    p[24000 : 24000 + len(burst)] = burst
    sigs["pulse_1k_70"] = p
    s = np.concatenate([sine(1000, 60, 0.5), sine(1000, 80, 0.5)])
    sigs["step_60_80"] = s
    # Annex B wav files (loaded exactly as mosqito tests do)
    for wav, key in [
        # v1.2.1 起 repo 只有 44.1 kHz 版；mosqito load() 會自動重採樣到 48 kHz
        ("Test signal 3 (1 kHz 60 dB)_44100Hz.wav", "annexb_sig3"),
        ("Test signal 5 (pinknoise 60 dB).wav", "annexb_sig5"),
        ("Test signal 10 (tone pulse 1 kHz 10 ms 70 dB).wav", "annexb_sig10"),
    ]:
        path = ROOT / "data" / "annexb" / wav
        if path.exists():
            sig, fs = load(str(path), wav_calib=2 * 2**0.5)
            assert fs == FS, f"{wav}: fs={fs}"
            sigs[key] = sig
    return sigs


def golden_zwst(d, sig, meta):
    spec_amp, _ = noct_spectrum(sig, FS, fmin=24, fmax=12600)
    meta["spec_third_amp"] = dump(d, "spec_third_amp", spec_amp)
    spec_db = amp2db(np.copy(spec_amp), ref=2e-5)
    meta["spec_third_db"] = dump(d, "spec_third_db", spec_db)
    nm = _main_loudness(spec_db, "free")
    meta["nm_free"] = dump(d, "nm_free", nm)
    nm_d = _main_loudness(spec_db, "diffuse")
    meta["nm_diffuse"] = dump(d, "nm_diffuse", nm_d)
    n, n_spec = _calc_slopes(nm)
    meta["N"] = dump(d, "N", np.atleast_1d(n))
    meta["N_specific"] = dump(d, "N_specific", n_spec)


def golden_zwtv(d, sig, meta):
    tol, time_axis, _ = _third_octave_levels(sig, FS)
    meta["third_octave_level"] = dump(d, "third_octave_level", tol)
    core = _main_loudness(tol, "free")
    meta["core_loudness"] = dump(d, "core_loudness", core)
    nl = _nl_loudness(np.copy(core))
    meta["nl_loudness"] = dump(d, "nl_loudness", nl)
    loud, spec_loud = _calc_slopes(nl)
    meta["loudness_raw"] = dump(d, "loudness_raw", loud)
    meta["spec_loudness_raw"] = dump(d, "spec_loudness_raw", spec_loud)
    filt = _temporal_weighting(loud)
    meta["filt_loudness"] = dump(d, "filt_loudness", filt)
    meta["N_time"] = dump(d, "N_time", filt[::4])
    meta["N_spec_time"] = dump(d, "N_spec_time", spec_loud[:, ::4])
    meta["time_axis"] = dump(d, "time_axis", time_axis[::4])


def golden_dsp(d, meta):
    """Unit-level DSP goldens: sosfilt / sosfiltfilt / decimate on a fixed input."""
    rng = np.random.default_rng(1)
    x = rng.standard_normal(4096)
    meta["dsp_x"] = dump(d, "dsp_x", x)
    sos = cheby1(8, 0.05, 0.8 / 10, output="sos")
    meta["dsp_cheby_sos"] = dump(d, "dsp_cheby_sos", sos)
    meta["dsp_sosfilt_y"] = dump(d, "dsp_sosfilt_y", sosfilt(sos, x))
    meta["dsp_sosfiltfilt_y"] = dump(d, "dsp_sosfiltfilt_y", sosfiltfilt(sos, x))
    meta["dsp_decimate_q10"] = dump(d, "dsp_decimate_q10", decimate(x, 10))


def main():
    OUT.mkdir(parents=True, exist_ok=True)
    d = OUT / "_dsp"
    d.mkdir(exist_ok=True)
    meta = {}
    golden_dsp(d, meta)
    (d / "meta.json").write_text(json.dumps(meta))

    for name, sig in make_signals().items():
        if sig is None:
            continue
        d = OUT / name
        d.mkdir(exist_ok=True)
        meta = {"fs": FS}
        meta["sig"] = dump(d, "sig", sig)
        golden_zwst(d, sig, meta)
        golden_zwtv(d, sig, meta)
        (d / "meta.json").write_text(json.dumps(meta))
        print("done", name)


if __name__ == "__main__":
    main()
```

```python
"""Generate iso532/src/tables_noct.rs: scipy-exact SOS coefficients for the
zwst 28-band 1/3-octave filter bank (ANSI S1.1 Butterworth order 3) and the
Chebyshev-I decimation filters, all fixed because fs is pinned to 48 kHz.
Run: .venv/Scripts/python tools/gen_tables.py
"""
from pathlib import Path

import numpy as np
from scipy.signal import butter, cheby1

FS = 48000
ROOT = Path(__file__).resolve().parent.parent


def center_freqs():
    # mosqito _center_freq(fmin=24, fmax=12600, n=3, G=10, fr=1000):
    # U = 10**(3*b/10) with b=1/3 -> U = 10**0.1
    u = 10 ** 0.1
    kmin = int(np.round(np.log10(24 / 1000) / np.log10(u)))
    kmax = int(np.round(np.log10(12600 / 1000) / np.log10(u)))
    return 1000 * u ** np.arange(kmin, kmax + 1)


def alpha(fc, n=3, N=3):
    b = 1 / n
    f1 = fc / (2 ** (b / 2))
    f2 = fc * (2 ** (b / 2))
    qr = fc / (f2 - f1)
    qd = (np.pi / 2 / N) / (np.sin(np.pi / 2 / N)) * qr
    return (1 + np.sqrt(1 + 4 * qd**2)) / 2 / qd


def decim_q(fc, fs=FS):
    if fc >= fs / 200:
        return 1
    q = 2
    while fc < fs / q / 200:
        q += 1
    return q


def fmt_sos(sos):
    rows = []
    for b0, b1, b2, a0, a1, a2 in sos:
        assert abs(a0 - 1.0) < 1e-15
        # float() strips numpy scalar wrappers; repr(float) round-trips exactly
        # and is valid Rust literal syntax (e.g. 0.001, 1e-05).
        vals = [repr(float(v)) for v in (b0, b1, b2, a1, a2)]
        rows.append(
            "    Sos { b: [%s, %s, %s], a: [%s, %s] }," % tuple(vals)
        )
    return "\n".join(rows)


def main():
    fcs = center_freqs()
    assert len(fcs) == 28, len(fcs)
    al = alpha(fcs)
    lines = [
        "//! GENERATED by tools/gen_tables.py -- do not edit by hand.",
        "//! scipy-exact SOS for the zwst noct_spectrum filter bank at fs=48kHz.",
        "use crate::dsp::sos::Sos;",
        "",
    ]
    qs, all_sos = [], []
    for fc, a in zip(fcs, al):
        q = decim_q(fc)
        fs_eff = FS / q
        w1 = fc / (fs_eff / 2) / a
        w2 = fc / (fs_eff / 2) * a
        sos = butter(3, (w1, w2), "bandpass", output="sos")
        qs.append(q)
        all_sos.append(sos)
    lines.append(f"pub const NOCT_DECIM_Q: [usize; 28] = {qs!r};".replace("'", ""))
    lines.append("")
    lines.append("pub const NOCT_SOS: [[Sos; 3]; 28] = [")
    for sos in all_sos:
        lines.append("  [")
        lines.append(fmt_sos(sos))
        lines.append("  ],")
    lines.append("];")
    lines.append("")
    uq = sorted(set(q for q in qs if q > 1))
    lines.append(f"pub const CHEBY_QS: [usize; {len(uq)}] = {uq!r};")
    lines.append(f"pub const CHEBY_SOS: [[Sos; 4]; {len(uq)}] = [")
    for q in uq:
        sos = cheby1(8, 0.05, 0.8 / q, output="sos")
        lines.append("  [")
        lines.append(fmt_sos(sos))
        lines.append("  ],")
    lines.append("];")
    out = ROOT / "iso532" / "src" / "tables_noct.rs"
    out.write_text("\n".join(lines) + "\n")
    print("wrote", out, "| qs =", qs)


if __name__ == "__main__":
    main()
```

```rust
#[derive(Debug, Clone, Copy, PartialEq)]
pub struct Sos {
    /// numerator [b0, b1, b2]
    pub b: [f64; 3],
    /// denominator [a1, a2] (a0 normalized to 1)
    pub a: [f64; 2],
}

/// Cascade of biquads, direct form II transposed, in place.
pub fn sosfilt(sections: &[Sos], x: &mut [f64]) {
    for s in sections {
        let (mut z0, mut z1) = (0.0f64, 0.0f64);
        for v in x.iter_mut() {
            let xin = *v;
            let y = s.b[0] * xin + z0;
            z0 = s.b[1] * xin - s.a[0] * y + z1;
            z1 = s.b[2] * xin - s.a[1] * y;
            *v = y;
        }
    }
}

/// Like sosfilt but with initial state `zi` (per section [z0, z1]),
/// returning the final state. Needed by sosfiltfilt.
pub fn sosfilt_zi_run(sections: &[Sos], x: &mut [f64], zi: &mut [[f64; 2]]) {
    for (s, z) in sections.iter().zip(zi.iter_mut()) {
        let (mut z0, mut z1) = (z[0], z[1]);
        for v in x.iter_mut() {
            let xin = *v;
            let y = s.b[0] * xin + z0;
            z0 = s.b[1] * xin - s.a[0] * y + z1;
            z1 = s.b[2] * xin - s.a[1] * y;
            *v = y;
        }
        *z = [z0, z1];
    }
}

/// scipy sosfilt_zi: steady state per section for unit step input,
/// scaled through the cascade by the cumulative DC gain.
pub fn sosfilt_zi(sections: &[Sos]) -> Vec<[f64; 2]> {
    let mut scale = 1.0;
    sections
        .iter()
        .map(|s| {
            let g = (s.b[0] + s.b[1] + s.b[2]) / (1.0 + s.a[0] + s.a[1]);
            let zi = [scale * (g - s.b[0]), scale * (s.b[2] - s.a[1] * g)];
            scale *= g;
            zi
        })
        .collect()
}

/// First-order lowpass y[n] = b0*x[n] + a1*y[n-1], in place.
pub fn onepole(b0: f64, a1: f64, x: &mut [f64]) {
    let mut y = 0.0f64;
    for v in x.iter_mut() {
        y = b0 * *v + a1 * y;
        *v = y;
    }
}
```

```rust
use super::sos::{sosfilt_zi, sosfilt_zi_run, Sos};
use crate::tables_noct::{CHEBY_QS, CHEBY_SOS};

/// Odd extension padding as scipy.signal._arraytools.odd_ext.
fn odd_ext(x: &[f64], n: usize) -> Vec<f64> {
    assert!(x.len() > n, "padlen {n} >= signal length {}", x.len());
    let mut out = Vec::with_capacity(x.len() + 2 * n);
    let x0 = x[0];
    for i in (1..=n).rev() {
        out.push(2.0 * x0 - x[i]);
    }
    out.extend_from_slice(x);
    let xl = x[x.len() - 1];
    for i in 2..=(n + 1) {
        out.push(2.0 * xl - x[x.len() - i]);
    }
    out
}

/// scipy.signal.sosfiltfilt with defaults (padtype='odd', padlen=None).
pub fn sosfiltfilt(sections: &[Sos], x: &[f64]) -> Vec<f64> {
    // scipy default: padlen = 3 * (2*n_sections + 1 - min(#(b2==0), #(a2==0)))
    let nb2 = sections.iter().filter(|s| s.b[2] == 0.0).count();
    let na2 = sections.iter().filter(|s| s.a[1] == 0.0).count();
    let padlen = 3 * (2 * sections.len() + 1 - nb2.min(na2));
    let zi_unit = sosfilt_zi(sections);

    let mut ext = odd_ext(x, padlen);
    // forward
    let x0 = ext[0];
    let mut zi: Vec<[f64; 2]> = zi_unit.iter().map(|z| [z[0] * x0, z[1] * x0]).collect();
    sosfilt_zi_run(sections, &mut ext, &mut zi);
    // backward
    ext.reverse();
    let y0 = ext[0];
    let mut zi: Vec<[f64; 2]> = zi_unit.iter().map(|z| [z[0] * y0, z[1] * y0]).collect();
    sosfilt_zi_run(sections, &mut ext, &mut zi);
    ext.reverse();
    ext[padlen..ext.len() - padlen].to_vec()
}

/// scipy.signal.decimate(x, q) with defaults: Chebyshev-I order 8, zero-phase.
/// Coefficients are baked per q in tables_noct (generated by scipy).
pub fn decimate(x: &[f64], q: usize) -> Vec<f64> {
    let idx = CHEBY_QS
        .iter()
        .position(|&c| c == q)
        .unwrap_or_else(|| panic!("no baked Chebyshev SOS for q={q}"));
    let y = sosfiltfilt(&CHEBY_SOS[idx], x);
    y.iter().step_by(q).copied().collect()
}
```

```rust
/// 1/3-octave band RMS levels, 28 bands (24 Hz..12.6 kHz), fs = 48 kHz.
/// Mirrors mosqito noct_spectrum for a single-channel signal.
pub fn noct_spectrum_rms(signal: &[f64]) -> Vec<f64> {
    (0..28)
        .map(|band| {
            let q = NOCT_DECIM_Q[band];
            let mut x = if q > 1 { decimate(signal, q) } else { signal.to_vec() };
            sosfilt(&NOCT_SOS[band], &mut x);
            let ms = x.iter().map(|v| v * v).sum::<f64>() / x.len() as f64;
            ms.sqrt()
        })
        .collect()
}

/// amp2db with ref 2e-5, zero amplitudes clamped like mosqito (2e-12).
pub fn spec_to_db(spec: &[f64]) -> Vec<f64> {
    spec.iter()
        .map(|&a| {
            let a = if a == 0.0 { 2e-12 } else { a };
            20.0 * (a / 2e-5).log10()
        })
        .collect()
}
```

```rust
use crate::tables::{A0, DCB, DDF, DLL, LTQ, RAP};
use crate::{FieldType, Iso532Error};

/// Core loudness for one frame of 28-band 1/3-octave levels [dB].
/// Returns 21 values (20 critical bands + trailing 0, as mosqito appends).
/// Ref: Zwicker/Fastl 1991 BASIC program; ISO 532-1:2017 A.3
/// (corr_third_octave_intensities, f_calc_lcbs, f_calc_core_loudness, f_corr_loudness).
pub fn main_loudness(spec_third: &[f64], field: FieldType) -> Result<[f64; 21], Iso532Error> {
    assert_eq!(spec_third.len(), 28);
    if spec_third[..11].iter().any(|&l| l > 120.0) {
        return Err(Iso532Error::LevelExceeds120dB);
    }

    // Low-frequency equal-loudness correction and intensities for bands < 315 Hz
    let mut ti = [0.0f64; 11];
    for i in 0..11 {
        let mut j = 0usize;
        while j < 7 && spec_third[i] > RAP[j] - DLL[j][i] {
            j += 1;
        }
        let xp = spec_third[i] + DLL[j][i];
        ti[i] = 10f64.powf(xp / 10.0);
    }

    // First three critical band levels from summed intensities
    let gi = [
        ti[0..6].iter().sum::<f64>(),
        ti[6..9].iter().sum::<f64>(),
        ti[9..11].iter().sum::<f64>(),
    ];
    let mut lcb = [0.0f64; 3];
    for k in 0..3 {
        if gi[k] > 0.0 {
            lcb[k] = 10.0 * gi[k].log10();
        }
    }

    // Core loudness per critical band
    const S: f64 = 0.25;
    let mut nm = [0.0f64; 21];
    for k in 0..20 {
        let mut le = if k < 3 { lcb[k] } else { spec_third[k + 8] };
        le -= A0[k];
        if field == FieldType::Diffuse {
            le += DDF[k];
        }
        if le > LTQ[k] {
            le -= DCB[k];
            let mp1 = 0.0635 * 10f64.powf(0.025 * LTQ[k]);
            let mp2 = (1.0 - S + S * 10f64.powf(0.1 * (le - LTQ[k]))).powf(0.25) - 1.0;
            nm[k] = (mp1 * mp2).max(0.0);
        }
    }
    // nm[20] stays 0 (mosqito appends a zero band)

    // Correction of the lowest critical band
    let korry = 0.4 + 0.32 * nm[0].powf(0.2);
    if korry <= 1.0 {
        nm[0] *= korry;
    }
    Ok(nm)
}
```

```rust
use crate::tables::{RNS, USL, ZUP};

/// mosqito compares with np.round(x, 8); replicate that tolerance semantic.
#[inline]
fn r8(x: f64) -> f64 {
    (x * 1e8).round() / 1e8
}

/// Loudness slopes + Bark integration for one frame of core loudness (21 bands).
/// Returns (total loudness N [sone], specific loudness at 240 z-steps of 0.1 Bark).
/// Structure follows the BASIC program of Zwicker/Fastl 1991 (DIN 45631) /
/// ISO 532-1:2017 C reference f_calc_slopes, not mosqito's vectorized rewrite.
pub fn calc_slopes(nm: &[f64; 21]) -> (f64, [f64; 240]) {
    let mut n_tot = 0.0f64;
    let mut ns = [0.0f64; 240];
    let mut n1 = 0.0f64; // loudness at left edge of current segment
    let mut z1 = 0.0f64; // critical band rate at left edge
    let mut iz = 0usize; // index into ns; grid point z = 0.1*(iz+1)

    // Band edges as grid indices: ns[..zup10[i]] belongs to bands 0..=i
    // (mosqito: zup_ea = (zup*10).astype(int))
    let zup10: [usize; 21] = std::array::from_fn(|i| (ZUP[i] * 10.0).round() as usize);

    for i in 0..21 {
        let zup_i = ZUP[i];
        let ig = if i == 0 { 0 } else { (i - 1).min(7) };

        loop {
            if r8(n1) <= r8(nm[i]) {
                // Rising or flat: rectangle up to the band edge
                let n2 = nm[i];
                n_tot += n2 * (zup_i - z1);
                while iz < zup10[i] {
                    ns[iz] = n2;
                    iz += 1;
                }
                z1 = zup_i;
                n1 = n2;
                break;
            } else {
                // Decreasing slope: steepness from RNS/USL lookup
                let mut j = 0usize;
                while j < 17 && RNS[j] >= n1 {
                    j += 1;
                }
                let usl = USL[j][ig];
                let mut n2 = RNS[j].max(nm[i]);
                let mut z2 = z1 + (n1 - n2) / usl;
                let clamped = z2 > zup_i;
                if clamped {
                    z2 = zup_i;
                    n2 = n1 - (z2 - z1) * usl;
                }
                n_tot += (z2 - z1) * (n1 + n2) / 2.0;
                while iz < zup10[i] {
                    let zg = 0.1 * (iz as f64 + 1.0);
                    if r8(zg) >= r8(z2) {
                        break;
                    }
                    ns[iz] = n1 - (zg - z1) * usl;
                    iz += 1;
                }
                z1 = z2;
                n1 = n2;
                if clamped {
                    // Slope reached the band edge: this band is done.
                    break;
                }
                // Otherwise the slope hit an RNS level or nm[i]; loop again
                // (another slope segment, or the rising branch closes the band).
            }
        }
    }

    // Final rounding as in the standard (and mosqito)
    let mut n = n_tot.max(0.0);
    n = if n <= 16.0 {
        (n * 1000.0 + 0.5).floor() / 1000.0
    } else {
        (n * 100.0 + 0.5).floor() / 100.0
    };
    (n, ns)
}
```

```rust
use crate::dsp::sos::{onepole, sosfilt, Sos};
use crate::tables::{TOB_DELTA, TOB_GAIN};

pub const DEC_FACTOR: usize = 24; // 48 kHz -> 2 kHz
const TINY: f64 = 1e-12;
const I_REF: f64 = 4e-10;

/// Reference-filter b coefficients per section (ISO Table A.1); the band-
/// dependent part only touches a1/a2 (Table A.2 deltas).
pub const TOB_B: [[f64; 3]; 3] = [[1.0, 2.0, 1.0], [1.0, 0.0, -1.0], [1.0, -2.0, 1.0]];

pub fn tob_sections(band: usize) -> [Sos; 3] {
    // std::array (not core::) -- the crate has its own `mod core`
    std::array::from_fn(|s| Sos {
        b: TOB_B[s],
        a: [-2.0 - TOB_DELTA[band][s][0], 1.0 - TOB_DELTA[band][s][1]],
    })
}

pub fn smoothing_coeff(band: usize) -> (f64, f64) {
    // center_freq = 10^((band-16)/10) * 1000 ; tau = 2/(3*min(fc,1000))
    let fc = 10f64.powf((band as f64 - 16.0) / 10.0) * 1000.0;
    let tau = 2.0 / (3.0 * fc.min(1000.0));
    let a1 = (-1.0 / (48000.0 * tau)).exp();
    (1.0 - a1, a1)
}

/// 28-band levels vs time at 2 kHz frame rate [dB].
/// Returns (row-major (28, n_time) buffer, n_time).
pub fn third_octave_levels_scalar(sig: &[f64]) -> (Vec<f64>, usize) {
    let n_time = sig.len().div_ceil(DEC_FACTOR); // len(sig[::24])
    let mut out = vec![0.0f64; 28 * n_time];
    for band in 0..28 {
        let mut x: Vec<f64> = sig.iter().map(|v| v * TOB_GAIN[band]).collect();
        sosfilt(&tob_sections(band), &mut x);
        for v in x.iter_mut() {
            *v *= *v;
        }
        let (b0, a1) = smoothing_coeff(band);
        for _ in 0..3 {
            onepole(b0, a1, &mut x);
        }
        for (t, v) in x.iter().step_by(DEC_FACTOR).enumerate() {
            out[band * n_time + t] = 10.0 * ((v + TINY) / I_REF).log10();
        }
    }
    (out, n_time)
}

/// time axis: np.linspace(0, len/fs, n_time)
pub fn time_axis(sig_len: usize, n_time: usize) -> Vec<f64> {
    let end = sig_len as f64 / 48000.0;
    (0..n_time).map(|i| end * i as f64 / (n_time - 1) as f64).collect()
}
```

注意 mosqito 是 `sig_filt = gain * sosfilt(coeff, sig)`（增益乘在濾波輸出）；上面乘在輸入端，對線性濾波器數學等價但浮點順序不同。若 golden rtol 1e-9 過不了，改成先濾波後乘增益完全復刻。

```rust
/// Constants B[0..6] of the two-capacitor analog network (ISO 532-1 / mosqito).
pub fn nl_coeffs() -> [f64; 6] {
    const SAMPLE_RATE: f64 = 2000.0;
    const NL_ITER: f64 = 24.0;
    const T_SHORT: f64 = 0.005;
    const T_LONG: f64 = 0.015;
    const T_VAR: f64 = 0.075;
    let delta_t = 1.0 / (SAMPLE_RATE * NL_ITER);
    let p = (T_VAR + T_LONG) / (T_VAR * T_SHORT);
    let q = 1.0 / (T_SHORT * T_VAR);
    let l1 = -p / 2.0 + (p * p / 4.0 - q).sqrt();
    let l2 = -p / 2.0 - (p * p / 4.0 - q).sqrt();
    let den = T_VAR * (l1 - l2);
    let e1 = (l1 * delta_t).exp();
    let e2 = (l2 * delta_t).exp();
    [
        (e1 - e2) / den,
        ((T_VAR * l2 + 1.0) * e1 - (T_VAR * l1 + 1.0) * e2) / den,
        ((T_VAR * l1 + 1.0) * e1 - (T_VAR * l2 + 1.0) * e2) / den,
        (T_VAR * l1 + 1.0) * (T_VAR * l2 + 1.0) * (e1 - e2) / den,
        (-delta_t / T_LONG).exp(),
        (-delta_t / T_VAR).exp(),
    ]
}

/// One inner step of the nonlinear lowpass (semantics = mosqito's mask logic,
/// which encodes the ISO C f_nl_lp truth table).
#[inline]
pub fn nl_lp_step(ui: f64, uo_last: f64, u2_last: f64, b: &[f64; 6]) -> (f64, f64) {
    let uo;
    let u2;
    if ui < uo_last {
        if uo_last > u2_last {
            // 2nd-order decay
            let uo_c = uo_last * b[2] - u2_last * b[3];
            uo = if uo_c >= ui { uo_c } else { ui };
            let u2_c = uo_last * b[0] - u2_last * b[1];
            u2 = if u2_c <= uo { u2_c } else { uo };
        } else {
            // 1st-order decay with t_long
            let uo_c = uo_last * b[4];
            uo = if uo_c >= ui { uo_c } else { ui };
            u2 = uo;
        }
    } else {
        // attack: output follows input
        uo = ui;
        if (ui - uo_last).abs() < 1e-5 && uo <= u2_last {
            u2 = uo;
        } else {
            u2 = (u2_last - ui) * b[5] + ui;
        }
    }
    (uo, u2)
}

/// Nonlinear temporal decay. Input/output: row-major (21, n_time) at 2 kHz.
/// 24 virtual upsampling steps with linear interpolation between frames.
pub fn nl_loudness_scalar(core: &[f64], n_time: usize) -> Vec<f64> {
    let b = nl_coeffs();
    let mut out = vec![0.0f64; 21 * n_time];
    for band in 0..21 {
        let row = &core[band * n_time..(band + 1) * n_time];
        let mut uo_last = 0.0f64;
        let mut u2_last = 0.0f64;
        for t in 0..n_time {
            let next = if t + 1 < n_time { row[t + 1] } else { 0.0 };
            let delta = (next - row[t]) / 24.0;
            let mut ui = row[t];
            for k in 0..24 {
                let (uo, u2) = nl_lp_step(ui, uo_last, u2_last, &b);
                uo_last = uo;
                u2_last = u2;
                if k == 0 {
                    out[band * n_time + t] = uo;
                }
                ui += delta;
            }
        }
    }
    out
}
```

已知差異風險（golden 失敗時的處置順序）：
1. **初始狀態**：mosqito 因 numpy 迴圈的 `col-1 = -1` 環繞，第一步的 `uo_last` 實際上是最後一個虛擬樣本值 `core[last]/24`、`u2_last=0`。若第一個 frame 對不上：把 `uo_last` 初值改為 `row[n_time-1]/24.0` 復刻該行為。
2. **比較運算子**：mask 是 `uo2 >= ui`（uo 取衰減值）與 `u22 <= uo`；上面已用 `>=`/`<=` 對齊，若仍差檢查 attack 分支的 `ui >= uo_last`（mosqito：`ui_delta >= uo_mat[col-1]`）。
3. mosqito 對 `uo` 的衰減 mask 沒檢查 `ui < uo_last`（僅 `uo_last > u2_last && uo2 >= ui`）；若邊界 frame 對不上，將 uo 的選擇改為完全復刻：先 `uo = ui`，若 `uo_last > u2_last && uo2 >= ui` 則 `uo = uo2`，再若 `uo_last <= u2_last && uo_last*b[4] >= ui` 則 `uo = uo_last*b[4]`，u2 邏輯同 mosqito 四段 mask 順序。

```rust
/// 1st-order lowpass with 24x linear interpolation (mosqito _lowpass_intp):
/// output is the filter state sampled at substep 0 of each frame.
fn lowpass_intp(x: &[f64], tau: f64) -> Vec<f64> {
    const SAMPLE_RATE: f64 = 2000.0;
    const LP_ITER: f64 = 24.0;
    let a1 = (-1.0 / (SAMPLE_RATE * LP_ITER * tau)).exp();
    let b0 = 1.0 - a1;
    let mut y = 0.0f64;
    let mut out = Vec::with_capacity(x.len());
    for t in 0..x.len() {
        let next = if t + 1 < x.len() { x[t + 1] } else { 0.0 };
        let delta = (next - x[t]) / 24.0;
        let mut ui = x[t];
        for k in 0..24 {
            y = b0 * ui + a1 * y;
            if k == 0 {
                out.push(y);
            }
            ui += delta;
        }
    }
    out
}

/// Duration-dependent loudness perception: 0.47*LP(3.5ms) + 0.53*LP(70ms).
pub fn temporal_weighting(loudness: &[f64]) -> Vec<f64> {
    let f1 = lowpass_intp(loudness, 3.5e-3);
    let f2 = lowpass_intp(loudness, 70e-3);
    f1.iter().zip(&f2).map(|(a, b)| 0.47 * a + 0.53 * b).collect()
}
```

```rust
pub fn loudness_zwtv(
    signal: &[f64],
    fs: f64,
    field: FieldType,
) -> Result<LoudnessTimeVarying, Iso532Error> {
    if fs != 48000.0 {
        return Err(Iso532Error::UnsupportedSampleRate(fs));
    }
    if signal.len() < 48 {
        return Err(Iso532Error::SignalTooShort { got: signal.len(), need: 48 });
    }
    let (tol, n_time) = third_octave_levels::third_octave_levels(signal);

    // core loudness per frame; keep (21, n_time) row-major
    let mut core = vec![0.0f64; 21 * n_time];
    let mut frame = [0.0f64; 28];
    for t in 0..n_time {
        for b in 0..28 {
            frame[b] = tol[b * n_time + t];
        }
        let nm = main_loudness(&frame, field)?;
        for b in 0..21 {
            core[b * n_time + t] = nm[b];
        }
    }

    let nl = nonlinear_decay::nl_loudness(&core, n_time);

    let mut loud = vec![0.0f64; n_time];
    let mut spec = vec![0.0f64; 240 * n_time]; // row-major (240, n_time)
    let mut nm = [0.0f64; 21];
    for t in 0..n_time {
        for b in 0..21 {
            nm[b] = nl[b * n_time + t];
        }
        let (n, ns) = calc_slopes(&nm);
        loud[t] = n;
        for z in 0..240 {
            spec[z * n_time + t] = ns[z];
        }
    }

    let filt = temporal_weighting::temporal_weighting(&loud);

    // decimate 0.5 ms -> 2 ms
    let n_out = n_time.div_ceil(4);
    let taxis = third_octave_levels::time_axis(signal.len(), n_time);
    let mut n_spec_out = vec![0.0f64; 240 * n_out];
    for z in 0..240 {
        for (o, t) in (0..n_time).step_by(4).enumerate() {
            n_spec_out[z * n_out + o] = spec[z * n_time + t];
        }
    }
    Ok(LoudnessTimeVarying {
        n: filt.iter().step_by(4).copied().collect(),
        n_specific: n_spec_out,
        bark_axis: bark_axis(),
        time_axis: taxis.iter().step_by(4).copied().collect(),
    })
}
```

```rust
/// AVX2+FMA kernel: 28 bands as 7 x f64x4, advancing sample by sample.
/// Layout per vector group v (0..7): bands 4v..4v+4.
/// # Safety
/// Caller must ensure AVX2 and FMA are available.
#[cfg(target_arch = "x86_64")]
#[target_feature(enable = "avx2,fma")]
pub unsafe fn third_octave_levels_avx2(sig: &[f64]) -> (Vec<f64>, usize) {
    use std::arch::x86_64::*;
    const NV: usize = 7; // 28 bands / 4 lanes

    let n_time = sig.len().div_ceil(DEC_FACTOR);
    let mut out = vec![0.0f64; 28 * n_time];

    // Pack per-band constants into lane vectors
    let mut gain = [_mm256_setzero_pd(); NV];
    let mut a1 = [[_mm256_setzero_pd(); NV]; 3]; // [section][vec]
    let mut a2 = [[_mm256_setzero_pd(); NV]; 3];
    let mut sb0 = [_mm256_setzero_pd(); NV]; // smoothing b0
    let mut sa1 = [_mm256_setzero_pd(); NV]; // smoothing a1
    for v in 0..NV {
        let mut g = [0.0f64; 4];
        let mut b0s = [0.0f64; 4];
        let mut a1s = [0.0f64; 4];
        let mut a1c = [[0.0f64; 4]; 3];
        let mut a2c = [[0.0f64; 4]; 3];
        for l in 0..4 {
            let band = 4 * v + l;
            g[l] = TOB_GAIN[band];
            let (b0, sa) = smoothing_coeff(band);
            b0s[l] = b0;
            a1s[l] = sa;
            for s in 0..3 {
                a1c[s][l] = -2.0 - TOB_DELTA[band][s][0];
                a2c[s][l] = 1.0 - TOB_DELTA[band][s][1];
            }
        }
        gain[v] = _mm256_loadu_pd(g.as_ptr());
        sb0[v] = _mm256_loadu_pd(b0s.as_ptr());
        sa1[v] = _mm256_loadu_pd(a1s.as_ptr());
        for s in 0..3 {
            a1[s][v] = _mm256_loadu_pd(a1c[s].as_ptr());
            a2[s][v] = _mm256_loadu_pd(a2c[s].as_ptr());
        }
    }
    // b coefficients per section are band-independent: [1,2,1],[1,0,-1],[1,-2,1]
    let b1s = [2.0f64, 0.0, -2.0];
    let b2s = [1.0f64, -1.0, 1.0];

    // States: 3 biquad sections (z0,z1) + 3 smoothing stages, per vector
    let mut z0 = [[_mm256_setzero_pd(); NV]; 3];
    let mut z1 = [[_mm256_setzero_pd(); NV]; 3];
    let mut sm = [[_mm256_setzero_pd(); NV]; 3];

    let mut frame = 0usize;
    for (i, &x) in sig.iter().enumerate() {
        let xs = _mm256_set1_pd(x);
        let store = i % DEC_FACTOR == 0;
        for v in 0..NV {
            // input gain
            let mut y = _mm256_mul_pd(xs, gain[v]);
            // 3 biquad sections, DF2T: y = b0*x + z0 (b0 = 1)
            for s in 0..3 {
                let xin = y;
                y = _mm256_add_pd(xin, z0[s][v]);
                // z0 = b1*x - a1*y + z1
                let b1v = _mm256_set1_pd(b1s[s]);
                let t = _mm256_fmadd_pd(b1v, xin, z1[s][v]);
                z0[s][v] = _mm256_fnmadd_pd(a1[s][v], y, t);
                // z1 = b2*x - a2*y
                let b2v = _mm256_set1_pd(b2s[s]);
                z1[s][v] = _mm256_fnmadd_pd(a2[s][v], y, _mm256_mul_pd(b2v, xin));
            }
            // squaring
            y = _mm256_mul_pd(y, y);
            // 3 smoothing one-poles: s = b0*y + a1*s
            for st in 0..3 {
                sm[st][v] = _mm256_fmadd_pd(sb0[v], y, _mm256_mul_pd(sa1[v], sm[st][v]));
                y = sm[st][v];
            }
            if store {
                let mut lanes = [0.0f64; 4];
                _mm256_storeu_pd(lanes.as_mut_ptr(), y);
                for l in 0..4 {
                    out[(4 * v + l) * n_time + frame] = lanes[l];
                }
            }
        }
        if store {
            frame += 1;
        }
    }
    // dB conversion (cheap, scalar): 28 * n_time elements
    for v in out.iter_mut() {
        *v = 10.0 * ((*v + TINY) / I_REF).log10();
    }
    (out, n_time)
}
```

```rust
/// AVX2 kernel: 21 bands in 6 x f64x4 (last vector has 3 padding lanes).
/// Branches become compare+blend; exact same decisions as nl_lp_step.
/// # Safety
/// Caller must ensure AVX2 and FMA are available.
#[cfg(target_arch = "x86_64")]
#[target_feature(enable = "avx2,fma")]
pub unsafe fn nl_loudness_avx2(core: &[f64], n_time: usize) -> Vec<f64> {
    use std::arch::x86_64::*;
    const NV: usize = 6; // ceil(21/4)
    let b = nl_coeffs();
    let (b0, b1, b2, b3, b4, b5) = (
        _mm256_set1_pd(b[0]), _mm256_set1_pd(b[1]), _mm256_set1_pd(b[2]),
        _mm256_set1_pd(b[3]), _mm256_set1_pd(b[4]), _mm256_set1_pd(b[5]),
    );
    let eps = _mm256_set1_pd(1e-5);
    let sign_mask = _mm256_set1_pd(-0.0);
    let inv24 = _mm256_set1_pd(1.0 / 24.0);

    let mut out = vec![0.0f64; 21 * n_time];
    let mut uo_last = [_mm256_setzero_pd(); NV];
    let mut u2_last = [_mm256_setzero_pd(); NV];

    // gather one frame's 21 values into padded [f64; 24]
    let load_frame = |t: usize| -> [f64; 24] {
        let mut f = [0.0f64; 24];
        for band in 0..21 {
            f[band] = core[band * n_time + t];
        }
        f
    };

    for t in 0..n_time {
        let cur = load_frame(t);
        let nxt = if t + 1 < n_time { load_frame(t + 1) } else { [0.0; 24] };
        for v in 0..NV {
            let mut ui = _mm256_loadu_pd(cur.as_ptr().add(4 * v));
            let nx = _mm256_loadu_pd(nxt.as_ptr().add(4 * v));
            let delta = _mm256_mul_pd(_mm256_sub_pd(nx, ui), inv24);
            for k in 0..24 {
                let uol = uo_last[v];
                let u2l = u2_last[v];
                // decay candidates
                let uo_d2 = _mm256_fnmadd_pd(b3, u2l, _mm256_mul_pd(uol, b2));
                let u2_d2 = _mm256_fnmadd_pd(b1, u2l, _mm256_mul_pd(uol, b0));
                let uo_d1 = _mm256_mul_pd(uol, b4);
                // masks
                let m_2nd = _mm256_cmp_pd(uol, u2l, _CMP_GT_OQ); // uo_last > u2_last
                let m_att = _mm256_cmp_pd(ui, uol, _CMP_GE_OQ); // ui >= uo_last
                // decay path
                let uo_dec_raw = _mm256_blendv_pd(uo_d1, uo_d2, m_2nd);
                let uo_dec = _mm256_max_pd(uo_dec_raw, ui);
                let u2_dec_2nd = _mm256_min_pd(u2_d2, uo_dec);
                let u2_dec = _mm256_blendv_pd(uo_dec, u2_dec_2nd, m_2nd);
                // attack path
                let u2_att_new = _mm256_fmadd_pd(_mm256_sub_pd(u2l, ui), b5, ui);
                let diff = _mm256_andnot_pd(sign_mask, _mm256_sub_pd(ui, uol)); // |ui-uo_last|
                let m_hold = _mm256_and_pd(
                    _mm256_cmp_pd(diff, eps, _CMP_LT_OQ),
                    _mm256_cmp_pd(ui, u2l, _CMP_LE_OQ),
                );
                let u2_att = _mm256_blendv_pd(u2_att_new, ui, m_hold);
                // select
                let uo = _mm256_blendv_pd(uo_dec, ui, m_att);
                let u2 = _mm256_blendv_pd(u2_dec, u2_att, m_att);
                uo_last[v] = uo;
                u2_last[v] = u2;
                if k == 0 {
                    let mut lanes = [0.0f64; 4];
                    _mm256_storeu_pd(lanes.as_mut_ptr(), uo);
                    for l in 0..4 {
                        let band = 4 * v + l;
                        if band < 21 {
                            out[band * n_time + t] = lanes[l];
                        }
                    }
                }
                ui = _mm256_add_pd(ui, delta);
            }
        }
    }
    out
}
```

**對齊備註**：scalar `nl_lp_step` 的 decay 分支是「先取候選、`>= ui` 才用否則 ui」＝`max(cand, ui)`；u2 的 2nd-order 分支是 `min(u2_cand, uo)`——上面用 max/min 表達。attack 的 hold 條件用 `uo(=ui) <= u2_last`。若 Task 11 曾為復刻 mosqito 而調整 scalar 邏輯（風險清單 1/3），此 kernel 的 blend 結構要跟著 scalar 的最終版同步改。

```rust
//! Compute ISO 532-1 loudness from a 48 kHz WAV file.
//! Usage: cargo run --example cli -- <file.wav> [--diffuse] [--time-varying] [--calib <factor>]

use iso532::{loudness_zwst, loudness_zwtv, FieldType};

fn main() {
    let args: Vec<String> = std::env::args().skip(1).collect();
    let Some(path) = args.iter().find(|a| !a.starts_with("--") && args.iter().position(|x| x == *a).map(|i| i == 0 || args[i-1] != "--calib").unwrap_or(true)).cloned() else {
        eprintln!("usage: cli <file.wav> [--diffuse] [--time-varying] [--calib <factor>]");
        std::process::exit(2);
    };
    let field = if args.iter().any(|a| a == "--diffuse") { FieldType::Diffuse } else { FieldType::Free };
    let calib: f64 = args.iter().position(|a| a == "--calib")
        .and_then(|i| args.get(i + 1)).and_then(|v| v.parse().ok()).unwrap_or(1.0);

    let mut reader = hound::WavReader::open(&path).expect("cannot open wav");
    let spec = reader.spec();
    assert_eq!(spec.sample_rate, 48000, "input must be 48 kHz (got {})", spec.sample_rate);
    let sig: Vec<f64> = match spec.sample_format {
        hound::SampleFormat::Int => {
            let denom = (1i64 << (spec.bits_per_sample - 1)) as f64;
            reader.samples::<i32>().map(|s| s.unwrap() as f64 / denom * calib).collect()
        }
        hound::SampleFormat::Float => reader.samples::<f32>().map(|s| s.unwrap() as f64 * calib).collect(),
    };

    if args.iter().any(|a| a == "--time-varying") {
        let r = loudness_zwtv(&sig, 48000.0, field).unwrap();
        let nmax = r.n.iter().cloned().fold(0.0f64, f64::max);
        println!("N_max = {nmax:.3} sone ({} frames @ 2 ms)", r.n.len());
        for (t, n) in r.time_axis.iter().zip(&r.n).step_by(50) {
            println!("{t:8.3} s  {n:8.3} sone");
        }
    } else {
        let r = loudness_zwst(&sig, 48000.0, field).unwrap();
        println!("N = {:.3} sone", r.n);
    }
}
```
